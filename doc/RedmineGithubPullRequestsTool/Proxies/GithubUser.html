def self.get_user_by_github_login(github_login)
  # Find the user or raise a RecordNotFound
  ::CustomValue.find_by_custom_field_id_and_value(
      self.github_login_name_field_id,
      github_login
  ).customized
end

def self.github_login_name_field_id
  unless ::Setting.respond_to? :redmine_github_pull_requests_tool
    raise Exceptions::PluginSettingsMissingError.new
  end
  settings = ::Setting.redmine_github_pull_requests_tool
  unless settings.key?(:github_login_name_field_id) && settings[:github_login_name_field_id] > 0
    raise Exceptions::PluginSettingsError.new(:github_login_name_field_id)
  end
  settings[:github_login_name_field_id]
end
