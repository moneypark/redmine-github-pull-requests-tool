def fg_color_for_html
  rgb = color.hex
  r = rgb >> 16
  g = (rgb & 65280) >> 8
  b = rgb & 255
  brightness = r * 0.299 + g * 0.587 + b * 0.114
  (brightness > 160) ? '#000' : '#fff'
end
